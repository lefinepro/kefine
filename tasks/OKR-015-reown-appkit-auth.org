#+TITLE: OKR-015: Reown AppKit Authentication
#+DATE: 2026-02-21
#+STARTUP: overview

* Overview
Integrate Reown AppKit for email and wallet authentication (Ethereum + Avalanche only). This provides users with seamless Web3 login options including email/social authentication and native wallet connections.

* Scope
** In Scope
- Email authentication via Reown AppKit
- Wallet connection (Ethereum + Avalanche chains only)
- Basic wallet connection UI component
- Authentication state management
- Session persistence

** Out of Scope
- Other blockchain networks (Solana, Bitcoin, etc.)
- Advanced DeFi features
- Token swaps or transfers
- NFT display

* Dependencies
- [[file:OKR-002-store-implementation.org][OKR-002-store-implementation.org]] - State management foundation
- [[file:OKR-013-activitypub-forgefed-integration.org][OKR-013-activitypub-forgefed-integration.org]] - User identity system (optional integration)

* Configuration Example

#+BEGIN_SRC typescript
// src/lib/auth/networks.ts
import { mainnet, avalanche } from '@reown/appkit/networks'

export const networks = [mainnet, avalanche]
#+END_SRC

#+BEGIN_SRC typescript
// src/lib/auth/appkit.ts
import { createAppKit } from '@reown/appkit/svelte'
import { networks } from './networks'

const projectId = import.meta.env.VITE_REOWN_PROJECT_ID

const metadata = {
  name: 'Kefine OKR Dashboard',
  description: 'OKR-focused task dashboard',
  url: window.location.origin,
  icons: ['https://kefine.app/icon.png']
}

export const appkit = createAppKit({
  adapters: [],
  networks,
  projectId,
  metadata,
  features: {
    analytics: true,
    email: true,
    socials: [] // Only email, no social providers
  },
  themeMode: 'light'
})
#+END_SRC

#+BEGIN_SRC svelte
// src/lib/components/auth/AuthButton.svelte
<script lang="ts">
  import { appkit } from '$lib/auth/appkit'
  import { authState } from '$lib/auth/auth-store'

  function handleConnect() {
    appkit.open()
  }

  function handleDisconnect() {
    appkit.disconnect()
  }
</script>

<button 
  class="auth-button"
  onclick={handleConnect}
  aria-label={authState.isConnected ? 'Disconnect wallet' : 'Connect wallet or email'}
>
  {#if authState.isConnected}
    <span class="connected-status">
      {#if authState.email}
        {authState.email}
      {:else if authState.address}
        {authState.address.slice(0, 6)}...{authState.address.slice(-4)}
      {/if}
    </span>
  {:else}
    Connect
  {/if}
</button>

<style>
  .auth-button {
    /* TAC styling */
  }
</style>
#+END_SRC
